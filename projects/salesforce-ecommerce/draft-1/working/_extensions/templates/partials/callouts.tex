% Callouts partial for Quarto Templates Framework
% Defines custom callout environments

% Success Story callout
\newenvironment{successstory}[1]{%
  \begin{tcolorbox}[
    enhanced,
    colback=successcolor!5,
    colframe=successcolor,
    fonttitle=\bfseries,
    title=#1,
    boxrule=0.5pt
  ]
}{%
  \end{tcolorbox}
}
