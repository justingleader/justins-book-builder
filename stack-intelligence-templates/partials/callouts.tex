% Note-style callout
\newenvironment{callout-note}[1]
{
\begin{calloutnote}{#1}
}
{
\end{calloutnote}
}

\newenvironment{callout-tip}[1]
{
\begin{callouttip}{#1}
}
{
\end{callouttip}
}

% Important-style callout
\newenvironment{callout-important}[1]
{
\begin{calloutimportant}{#1}
}
{
\end{calloutimportant}
}
